│ 🛠️  COMANDOS ÚTILES                                                         │
└──────────────────────────────────────────────────────────────────────────────┘

INSTALACIÓN
   iwr https://railway.app/install.ps1 | iex    # Instalar Railway CLI
   railway --version                             # Verificar instalación

CONFIGURACIÓN
   railway login                                 # Iniciar sesión
   railway link                                  # Vincular proyecto
   railway status                                # Ver estado

DESPLIEGUE
   railway run python manage.py migrate         # Ejecutar migraciones
   railway run python manage.py createsuperuser # Crear superusuario
   railway logs                                  # Ver logs

BASE DE DATOS
   railway run psql $DATABASE_URL                # Conectar a PostgreSQL
   .\export_database.ps1                         # Exportar DB local
   .\import_to_railway.ps1                       # Importar a Railway
